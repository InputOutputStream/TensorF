#include <iostream>
#include <iomanip> // Required for manipulators

int main() {
    double pi = 3.14159265359;

    std::cout << "Default precision: " << pi << std::endl;
    std::cout << "Precision set to 4: " << std::setprecision(4) << pi << std::endl;
    std::cout << "Precision set to 2 with fixed notation: " << std::fixed << std::setprecision(2) << pi << std::endl;
    std::cout << "Precision set to 3 with scientific notation: " << std::scientific << std::setprecision(3) << pi << std::endl;

    // Reset to default settings
    std::cout << std::defaultfloat; // Reset floating-point format to default
    std::cout << std::setprecision(6); // Reset precision to default
    std::cout << "Default precision and format: " << pi << std::endl;

    return 0;
}


#include <iostream>

int main() {
    std::ostream& out = std::cout; // Alias for std::cout

    out.put('H');
    out.put('i');
    out.put('\n');

    const char* message = "Hello, world!";
    out.write(message, 13); // Write the first 13 characters of the message
    out << std::endl; // Flush the output buffer and insert a newline

    out.flush(); // Flush the output buffer explicitly

    return 0;
}


template <typename E>
ostream& operator<<(ostream& out, const Matrix<E>& m) {
    out << "Values:" << endl;
    for (const auto& row : m.val) {
        for (const auto& element : row) {
            out << element << "\t";
        }
        out << endl;
    }
    out << "Shape: " << m.shape.first << "x" << m.shape.second << endl;

    return out;
}
